    • Get the number and range from the user using for loop 
    
                  
    

number = int(input("Enter a number: "))
range_limit = int(input("Enter the range (number of multiples to print): "))

print(f"Multiplication table for {number} up to {range_limit} multiples:")  
for i in range(1, range_limit + 1):
print(f"{number} x {i} = {number * i}")
    • Get the number and range from the user using While loop 
    
                  
    
                    
number = int(input("Enter a number: "))
range_limit = int(input("Enter the range (number of multiples to print): "))
                    
i = 1
print(f"Multiplication table for {number} up to {range_limit} multiples:")
while i <= range_limit:
print("{number} x {i} = {number * i}")
i += 1
    • Get the number of rows from the user using nested loop 
    
                  
    
                    
                    
num_rows = int(input("Enter the number of rows: "))

print("Pattern of Asterisks:")
for i in range(num_rows):
    for j in range(i + 1):
        print("*", end=" ")
    print()  # Move to the next line after printing a row
